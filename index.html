container.classList.add('message'); // Username const username = document.createElement('div'); username.classList.add('username'); username.textContent = msg.username || 'Unbekannt'; // Nachrichtentext const messageText = document.createElement('div'); messageText.textContent = msg.content; // Timestamp generieren const timestamp = document.createElement('div'); timestamp.classList.add('timestamp'); const date = msg.created_at ? new Date(msg.created_at) : new Date(); const timeString = date.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }); timestamp.textContent = timeString; // Zusammenbauen → jetzt inkl. Username! container.appendChild(username); container.appendChild(messageText); container.appendChild(timestamp); messagesDiv.appendChild(container); } // Reload-Button btnReload.addEventListener('click', loadMessages) function startAutoReload() { if (reloadInterval) clearInterval(reloadInterval) reloadInterval = setInterval(loadMessages, 1000) // jede Sekunde reload } // Nachricht senden async function sendMessage() { const text = msgInput.value.trim() if (!text) return const { error } = await supabaseClient.from('messages').insert([ { id: crypto.randomUUID(), username: currentUser.username, content: text } ]) if (error) { console.error('Fehler beim Senden:', error) alert('Fehler beim Senden der Nachricht') return } msgInput.value = '' loadMessages() // direkt reload nach senden } btnSend.addEventListener('click', sendMessage) // ENTER im Chat-Eingabefeld msgInput.addEventListener('keydown', e => { if (e.key === 'Enter') { e.preventDefault() sendMessage() } }) // Logout btnLogout.addEventListener('click', () => { currentUser = null localStorage.removeItem('chatUser') chatDiv.style.display = 'none' loginDiv.style.display = 'block' messagesDiv.innerHTML = '' clearInterval(reloadInterval) }) // Settings-Menü toggle btnSettings.addEventListener('click', () => { settingsMenu.style.display = settingsMenu.style.display === 'block' ?
